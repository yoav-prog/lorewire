# LoreWire pipeline.
#
# The pipeline was originally stdlib-only, but the Google Cloud REST APIs
# require a service-account JWT signed with RSA-SHA256 — Python stdlib has
# no RSA primitives. google-auth gives us a signer + JWT encoder; HTTP is
# still done with stdlib urllib.
#
# Install:  python -m pip install -r pipeline/requirements.txt

google-auth>=2.50,<3
# Production Postgres (admin on Vercel). Activated by DATABASE_URL; SQLite stays
# the default when the env var is absent. The binary extra ships compiled libpq
# so no separate system install is needed on Windows/macOS/Linux.
psycopg[binary]>=3.1,<4
# Number-to-words for the caption text normalizer (pipeline/text_normalize.py).
# Pre-normalizing the script before TTS keeps the spoken audio and the caption
# text in lockstep: "$1,000,000" becomes "one million dollars" in both, so the
# karaoke highlight matches what the voice is saying. MIT licensed, pure
# Python, one transitive dep (docopt for its unused CLI).
num2words>=0.5,<1
